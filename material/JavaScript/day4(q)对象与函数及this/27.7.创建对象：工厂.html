<script> /* * 创建n个对象 * */ var obj1= { name:'Victor', age:24, gender:'Male', sayName:function(){ alert(this.name); } }; /* * 使用工厂方法创建对象 * */ function iObject(name , age , gender){ //创建一个新对象 var obj = new Object(); obj.name = name, obj.age = age, obj.gender = gender, obj.sayName = function(){ alert(this.name); } //将新的对象返回 return obj; } var obj1 = iObject("Victor",24,"Male"); var obj2 = iObject("Doris",22,"Female"); var obj3 = iObject("Luke",22,"Male"); var obj4 = iObject("Wayne",19,"Male"); /* * 使用工厂方法创建的对象，使用的构造函数都是Object * 所以创建的对象都是Object这个类型， * 就导致我们无法区分出多种不同类型的对象 * */ </script>
