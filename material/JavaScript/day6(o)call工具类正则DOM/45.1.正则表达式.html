<script> /* * 正则表达式 * - admin * - admin@atguigu.com * - admin@.com adminatguigu.com * - 邮件的规则： * 1.xxxx * 2.@ * 3.xxxx * 4..com * - 正则表达式用于定义一些字符串规则， * 计算机根据正则表达式，检查一个字符串是否符合规则， * 或者将字符串中符合规则的内容提取出来 * */ //创建正则表达式对象 /* * 语法： * var 变量 = new RegExp("正则表达式","匹配模式"); * 使用typeof检查正则对象，会返回object * */ var reg = new RegExp("a"); var str = "a"; /* * 正则表达式的方法： * test() * - 用来检查字符串是否符合正则表达式的规则， * 符合 -> 返回true ; 不符合 -> 返回false * - 参数： * Fir.用来检查是否有 * Sec.传一个参数作为匹配模式 * i: 忽略大小写 * g: 全局匹配模式 * */ var result = reg.test(str); console.log(reg.test("abc")); </script>
