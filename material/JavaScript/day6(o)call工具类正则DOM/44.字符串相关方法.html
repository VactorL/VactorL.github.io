<script> //创建一个字符串 var str = "Hello VactorL"; /* * 在底层字符串是以字符数组的形式保存的 * ["H" , "e" , "l" , "l" , "o" , " " , "V" , "a" , "c" , "t" , "o" , "r" , "L"] * */ /* * length属性 * - 获取字符串的长度 * */ //console.log(str.length); //console.log(str[5]); /* * charAt() * - 返回字符串中指定位置的字符 * - 根据索引获取指定字符 * */ var result = str.charAt(6); /* * charCodeAt() * - 获取指定位置字符的字符编码(Unicode编码) * */ result = str.charCodeAt(3); /* * String.fromCharCode() * - 根据字符编码获取字符 * */ result = String.fromCharCode(20234); /* * concat() * - 连接两个或多个字符串 * - 作用和 + 一样 * */ result = str.concat(" GG blayt"); /* * indexof() * - 检索一个字符串中是否含有指定内容 * - 如果字符串中含有该内容，则会返回其第一次出现的索引 * 如果没有找到指定的内容，则返回-1 * - 可以指定第二个参数，指定开始查找的索引位置 * lastIndexof(); * - 用法与indexof()相同 * 不同点：查找顺序相反 * indexof()从前往后 * lastIndexof()从后往前 * */ /* * slice() * - 从字符串中截取指定的内容 * - 不影响原字符串，而是将截取到的内容返回 * - 参数： * Fir.开始位置的索引(包括开始位置) * Sec.结束位置的索引(不包括结束位置) * - 若省略Sec.则截取后面所有 * - 也可以传递一个负数，表示从后往前计算 * */ /* * substring() * - 用来截取一个字符串，与slice()类似 * - 参数： * Fir.开始位置的索引(包括开始位置) * Sec.结束位置的索引(不包括结束位置) * - 该方法不能接受负值 * - 且会自动调整参数位置，若Sec.>Fir.则自动交换 * */ /* * substr() * - 用来截取字符串 * - 参数 * Fir.截取开始位置 * Sec.截取长度 * */ console.log(result); /* * split() * - 可以将一个字符串拆分为一个数组 * - 参数： * - 以一个字符串作为参数，根据该字符串拆分数组 * */ str = "abc,def,ghi,jkl,mn,o"; result = str.split(","); console.log(result); /* * toUpperCase() * - 变大写，返回 * toLowerCase() * - 变小写，返回 * */ </script>
