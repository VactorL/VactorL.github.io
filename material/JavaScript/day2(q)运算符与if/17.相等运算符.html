<script> /* * 相等运算符：用于比较两个值是否相等， * 相等 -> 返回true，不相等 -> 返回false * * 使用 == 来做相等运算 * - 当时用 == 比较两个值时，若类型不同， * 则会自动进行类型转换，将其转换成为相同类型 * 然后再比较 * ※具体转换成什么类型不确定，大多数会转换成Number * - NaN不和任何值相等，包括NaN本身 * * 不相等运算符： * 使用 != 来做不相等运算 * - 运算规则与 == 相同 * * === * 全等 * - 用来判断两个值是否全等，和相等类似，但不做类型转换 * 如果两个值类型不同，直接返回false * * !== * 不全等 * - 用来判断两个值是否不全等，和不相等类似，但不做类型转换 * 如果两个值类型不同，直接返回true * * * */ var b = NaN; console.log('123' == 123); console.log('123' != 123); console.log('123' === 123); console.log('123' !== 123); console.log(isNaN(b)); //isNan()函数：检查NaN </script>
