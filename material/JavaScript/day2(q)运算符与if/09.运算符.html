<script> /* * 运算符，也叫操作符 * 通过运算符可对一个或多个值进行计算 * 比如：typeof就是运算符，可以来获得一个值的类型 * 会将该值得类型以字符串的形式返回 * Number String Boolean Undefined Object * * 算术运算符 * + 加 * 任何值和字符串做加法运算，都会先转换为字符串，然后再和字符串做拼串的操作 * 可以利用这一点将任意数据类型转换为字符串 * 只需要为任意类型 + 一个 "" * 从左到右依次运算 * - 减 * 乘 / 除 * 任何值做这三种运算都会自动转换为Number * 可以通过为一个值 -0 *1 /1 来将其转换为Number * % 取模运算(取余数) * */ var a = null; var result = typeof a; a = "123" + "456"; //拼串 b = true - "123"; c = null + "null" + undefined; console.log('a = ' + a); console.log('b = ' + b); console.log('c = ' + c); console.log('a.type: ' + typeof a); d = 123; d = d + ""; e = 2 * "8"; f = "123"; f = f - 0; console.log('d = ' + d); console.log('e = ' + e); console.log('f = ' + f); // console.log('g = ' + g); console.log('f.type: ' + typeof f); </script>
